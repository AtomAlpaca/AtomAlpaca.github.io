#include <bits/stdc++.h>

using std::cin;
using std::cout;

typedef long long ll;
const ll MAX = 4e4 + 5;
ll n, f[MAX];

int main()
{
    cin >> n; --n; if (!n) { cout << 0; return 0; }
    for (ll i = n; i >= 1; --i)
    {
        f[i] = (n / i) * (n / i);
        for (ll j = 2; j * i <= n; ++j) { f[i] -= f[i * j]; }
    }
    cout << f[1] + 2;
    return 0;
}
